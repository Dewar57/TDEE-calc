    <script>
        // 1. Load data from iOS Local Storage
        let logs = JSON.parse(localStorage.getItem('tdeeLogs')) || [];

        // Set today's date automatically in the input
        document.getElementById('dateInput').valueAsDate = new Date();

        // 2. The Core Math Logic
        function calculateAlgorithm() {
            // Sort by date from oldest to newest
            logs.sort((a, b) => new Date(a.date) - new Date(b.date));

            for (let i = 0; i < logs.length; i++) {
                let current = logs[i];
                
                // Handle missing calories: Use median of last 7 days if blank
                let effectiveCal = current.cal;
                if (!effectiveCal && i > 0) {
                    let pastCals = logs.slice(Math.max(0, i-7), i).map(l => l.effectiveCal).filter(c => c > 0);
                    if (pastCals.length > 0) {
                        pastCals.sort((a,b) => a - b);
                        effectiveCal = pastCals[Math.floor(pastCals.length / 2)]; // Median
                    } else {
                        effectiveCal = 2000; // Fallback baseline
                    }
                }
                current.effectiveCal = parseFloat(effectiveCal) || 0;

                // First day baseline setup
                if (i === 0) {
                    current.trendWt = current.wt;
                    current.trendCal = current.effectiveCal;
                    current.tdee = null;
                } else {
                    // Exponential Moving Average (10% today, 90% history)
                    let prev = logs[i-1];
                    current.trendWt = (current.wt * 0.1) + (prev.trendWt * 0.9);
                    current.trendCal = (current.effectiveCal * 0.1) + (prev.trendCal * 0.9);

                    // Calculate TDEE if we have at least 7 days of history
                    if (i >= 7) {
                        let wtDiff = current.trendWt - logs[i-7].trendWt; // Change over last week
                        let dailyWtChange = wtDiff / 7;
                        // 3500 kcal per lb of tissue. (Change to 7700 if using KGs)
                        current.tdee = current.trendCal - (dailyWtChange * 3500); 
                    } else {
                        current.tdee = null;
                    }
                }
            }
            renderUI();
        }

        // 3. Save User Input
        function logData() {
            const date = document.getElementById('dateInput').value;
            const wt = parseFloat(document.getElementById('weightInput').value);
            const cal = document.getElementById('calInput').value ? parseFloat(document.getElementById('calInput').value) : null;

            if (!date || isNaN(wt)) { alert("Please enter at least the date and weight."); return; }

            // Update if date exists, else push new
            const existingIndex = logs.findIndex(l => l.date === date);
            if (existingIndex > -1) {
                logs[existingIndex].wt = wt;
                logs[existingIndex].cal = cal;
            } else {
                logs.push({ date: date, wt: wt, cal: cal });
            }

            localStorage.setItem('tdeeLogs', JSON.stringify(logs));
            document.getElementById('weightInput').value = '';
            document.getElementById('calInput').value = '';
            calculateAlgorithm();
        }

        // 4. Update the Screen
        function renderUI() {
            const tbody = document.getElementById('historyTable');
            tbody.innerHTML = '';
            
            if (logs.length === 0) return;

            // Update Top Dashboard
            const latest = logs[logs.length - 1];
            document.getElementById('displayTrendWt').innerText = latest.trendWt.toFixed(1);
            
            if (latest.tdee) {
                document.getElementById('displayTdee').innerText = Math.round(latest.tdee) + " kcal";
            } else {
                document.getElementById('displayTdee').innerText = "Need " + (8 - logs.length) + " more days";
            }

            // Populate Table (Newest first)
            let reversedLogs = [...logs].reverse();
            reversedLogs.forEach(log => {
                let tr = document.createElement('tr');
                tr.innerHTML = `
                    <td>${log.date.substring(5)}</td>
                    <td>${log.wt}</td>
                    <td>${log.cal || '-'}</td>
                    <td>${log.trendWt.toFixed(1)}</td>
                    <td>${log.tdee ? Math.round(log.tdee) : '-'}</td>
                `;
                tbody.appendChild(tr);
            });
        }

        // Run on load
        calculateAlgorithm();
    </script>
